rules_version = '2';
service firebase.storage {
  match /b/{bucket}/o {
    // Allow authenticated users to read any file
    match /{allPaths=**} {
      allow read: if request.auth != null;
    }
    
    // Profile images
    match /profile_images/{userType}/{userId}/{fileName} {
      // Allow users to upload their own profile images
      allow write: if request.auth != null && request.auth.uid == userId;
    }
    
    // Document images (medical licenses, degrees, CNIC, etc.)
    match /documents/{userType}/{userId}/{documentType} {
      // Allow users to upload their own documents
      allow write: if request.auth != null && request.auth.uid == userId;
    }
    
    // General image uploads
    match /images/{userId}/{fileName} {
      // Allow users to upload their own images
      allow write: if request.auth != null && request.auth.uid == userId;
    }
  }
}
